## Tauon Music Box config file

## Fade times in ms. Full gapless playback not supported (min 50ms, max 3000ms)
pause-fade-time=400
cross-fade-time=700
scrobble-mark=True
scroll-gallery-row=False
scroll-gallery-wheel=100

## Font files must be in /gui subfolder
font-regular=Koruri-Regular.ttf
font-fallback=DroidSansFallback.ttf
font-bold=Koruri-Semibold.ttf
font-extra=Koruri-Light.ttf
font-height-offset=0

## Media keys; 0 = off, 1 = source from gnome/dbus, 2 = source from x server (requires pyxhook; sudo pip3 install pyxhook)
mediakey=2

## Output directory for encoded files. Must have user permissions. Default is encoder folder in program directory.
#output-dir=C:\example

## Icecast mounting. Codec can be MP3, OGG
## Note: MP3 setting requires lame. On windows place 'lame.exe' in /encoder directory. On Linux install 'lame' package
## Tip: Higher bitrates may help reduce latency

icecast-codec=OGG
icecast-bitrate=128
icecast-mount=localhost:8000/test
icecast-password=hackme

## External Tag editor loading. tag-editor-program is the command used on Linux, tag-editor-path is filepath to the exe used on Windows (exclude quotes). When program is closed, tags will be updated provided the filenames of tracks have not changed.
tag-editor-name=Picard
tag-editor-program=picard
tag-editor-path=C:/Program Files (x86)/MusicBrainz Picard/picard.exe
